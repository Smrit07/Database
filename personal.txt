-- Main Table
CREATE TABLE Personal_Details (
    F_Name VARCHAR2(30),
    L_Name VARCHAR2(30),
    Date_of_Birth DATE,
    Gender VARCHAR2(8),
    Birth_Certificate_no NUMBER(10)
);

-- Contact Table
CREATE TABLE Contact_Details (
    Phone_no NUMBER(15),
    Email VARCHAR2(20),
    T_Address VARCHAR2(20),
    P_Address VARCHAR2(20),
    Birth_Certificate_no NUMBER(10)
);

-- Qualification Table
CREATE TABLE Qualification (
    Primary_level VARCHAR2(30),
    Secondary_level VARCHAR2(30),
    Higher_level VARCHAR2(30),
    Registration_no VARCHAR2(20),
    Birth_Certificate_no NUMBER(10)
);

-- Medical Table
CREATE TABLE Medical_Info (
    Blood_group VARCHAR2(5),
    Medical_id NUMBER(10),
    Birth_Certificate_no NUMBER(10)
);
